<structure><name> Adiposity </name>

<variables>

<constant>
<name> MALE_UNDERWEIGHT </name>
<val> 10.0 </val>
</constant>

<constant>
<name> MALE_NORMAL </name>
<val> 17.0 </val>
</constant>

<constant>
<name> MALE_OVERWEIGHT </name>
<val> 26.0 </val>
</constant>

<constant>
<name> MALE_OBESE </name>
<val> 44.0 </val>
</constant>

<constant>
<name> MALE_MORBIDLYOBESE </name>
<val> 110.0 </val>
</constant>

<constant>
<name> FEMALE_UNDERWEIGHT </name>
<val> 14.0 </val>
</constant>

<constant>
<name> FEMALE_NORMAL </name>
<val> 20.0 </val>
</constant>

<constant>
<name> FEMALE_OVERWEIGHT </name>
<val> 32.0 </val>
</constant>

<constant>
<name> FEMALE_OBESE </name>
<val> 46.0 </val>
</constant>

<constant>
<name> FEMALE_MORBIDLYOBESE </name>
<val> 110.0 </val>
</constant>

<parm><name> Mass(kG) </name><val> 16.6 </val></parm>

<var><name> Mass </name></var>
<var><name> Index </name><val> 1 </val></var>

<var><name> IsUnderweight </name><val> FALSE </val></var>
<var><name> IsNormalWeight </name><val> TRUE </val></var>
<var><name> IsOverweight </name><val> FALSE </val></var>
<var><name> IsObese </name><val> FALSE </val></var>
<var><name> IsMorbidlyObese </name><val> FALSE </val></var>

</variables>

<definitions>

<block><name> Parms </name>

<onjustchanged><name> Mass(kG) </name>
<call> Update </call>

<if><test> Gender.IsFemale </test>
<true><call> IndexFemale </call></true>
<false><call> IndexMale </call></false>
</if>

<call> Context.Size </call>
</onjustchanged>

</block>

<block><name> Update </name>

<def>
<name> Mass </name>
<val> 1000 * Mass(kG) </val>
</def>

</block>

<block><name> Normalize </name>

<if><test> Gender.IsFemale </test>
<true><call> NormalizeFemale </call></true>
<false><call> NormalizeMale </call></false>
</if>

<call> Update </call>

</block>

<block><name> IndexMale </name>

<testcase>

<case><test> Mass(kG) LT ( ( MALE_UNDERWEIGHT + MALE_NORMAL ) / 2 ) </test>
<def><name> Index </name><val> 0 </val></def>
</case>

<case><test> Mass(kG) LT ( ( MALE_NORMAL + MALE_OVERWEIGHT ) / 2 ) </test>
<def><name> Index </name><val> 1 </val></def>
</case>

<case><test> Mass(kG) LT ( ( MALE_OVERWEIGHT + MALE_OBESE ) / 2 ) </test>
<def><name> Index </name><val> 2 </val></def>
</case>

<case><test> Mass(kG) LT ( ( MALE_OBESE + MALE_MORBIDLYOBESE ) / 2 ) </test>
<def><name> Index </name><val> 3 </val></def>
</case>

<case><test> TRUE </test>
<def><name> Index </name><val> 4 </val></def>
</case>

</testcase>

</block>

<block><name> IndexFemale </name>

<testcase>

<case><test> Mass(kG) LT ( ( FEMALE_UNDERWEIGHT + FEMALE_NORMAL ) / 2 ) </test>
<def><name> Index </name><val> 0 </val></def>
</case>

<case><test> Mass(kG) LT ( ( FEMALE_NORMAL + FEMALE_OVERWEIGHT ) / 2 ) </test>
<def><name> Index </name><val> 1 </val></def>
</case>

<case><test> Mass(kG) LT ( ( FEMALE_OVERWEIGHT + FEMALE_OBESE ) / 2 ) </test>
<def><name> Index </name><val> 2 </val></def>
</case>

<case><test> Mass(kG) LT ( ( FEMALE_OBESE + FEMALE_MORBIDLYOBESE ) / 2 ) </test>
<def><name> Index </name><val> 3 </val></def>
</case>

<case><test> TRUE </test>
<def><name> Index </name><val> 4 </val></def>
</case>

</testcase>

</block>

<block><name> NormalizeMale </name>

<testcase>

<case><test> Index EQ 0 </test>
<def><name> Mass(kG) </name><val> MALE_UNDERWEIGHT </val></def>
</case>

<case><test> Index EQ 1 </test>
<def><name> Mass(kG) </name><val> MALE_NORMAL </val></def>
</case>

<case><test> Index EQ 2 </test>
<def><name> Mass(kG) </name><val> MALE_OVERWEIGHT </val></def>
</case>

<case><test> Index EQ 3 </test>
<def><name> Mass(kG) </name><val> MALE_OBESE </val></def>
</case>

<case><test> Index EQ 4 </test>
<def><name> Mass(kG) </name><val> MALE_MORBIDLYOBESE </val></def>
</case>

</testcase>

</block>

<block><name> NormalizeFemale </name>

<testcase>

<case><test> Index EQ 0 </test>
<def><name> Mass(kG) </name><val> FEMALE_UNDERWEIGHT </val></def>
</case>

<case><test> Index EQ 1 </test>
<def><name> Mass(kG) </name><val> FEMALE_NORMAL </val></def>
</case>

<case><test> Index EQ 2 </test>
<def><name> Mass(kG) </name><val> FEMALE_OVERWEIGHT </val></def>
</case>

<case><test> Index EQ 3 </test>
<def><name> Mass(kG) </name><val> FEMALE_OBESE </val></def>
</case>

<case><test> Index EQ 4 </test>
<def><name> Mass(kG) </name><val> FEMALE_MORBIDLYOBESE </val></def>
</case>

</testcase>

</block>

</definitions>
</structure>
