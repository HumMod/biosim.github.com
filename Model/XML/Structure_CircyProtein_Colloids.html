<variables>

<var>
<name> Pres </name>
</var>

<var>
<name> [Prot] </name>

<constant>
<name> C1 </name>
<val> 320.0 </val>
</constant>

<constant>
<name> C2 </name>

</constant>

</constant>

<definitions>
<block><name> GetPres </name>

<conditional>
<name> Pres </name>
<test> [Prot] GT 0.0 </test>
<true> ( C1 * [Prot] ) + ( C2 * [Prot] * [Prot] ) </true>
<false> 0.0 </false>
</conditional>

</block>

<block><name> Get[Prot] </name>

<conditional>
<name> [Prot] </name>

<true> SQRT ( K2 + ( Pres / C2 ) ) - K1
</true>
<false> 0.0 </false>
</conditional>

</block>
</definitions>
