( ( 1.48 * Fat-Size.InitialMass(kG) )

<def>
<name> Fat_kCal/Day </name>

( ( 1.32 * Fat-Size.InitialMass(kG) )

) ROUNDAT 10
</val>
</def>

<def>
<name> Protein_kCal/Day </name>

( ( 0.82 * Fat-Size.InitialMass(kG) )

) ROUNDAT 10
</val>
</def>

</block>
