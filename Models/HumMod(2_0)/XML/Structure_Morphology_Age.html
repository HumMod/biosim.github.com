<structure><name> Age </name>

<variables>

<constant>
<name> YOUNG </name>
<val> 19.0 </val>
</constant>

<constant>
<name> YOUNGISH </name>
<val> 28.0 </val>
</constant>

<constant>
<name> MIDDLEAGED </name>
<val> 37.0 </val>
</constant>

<constant>
<name> OLD </name>
<val> 65.0 </val>
</constant>

<constant>
<name> VERYOLD </name>
<val> 89.0 </val>
</constant>

<parm><name> InitialAge(Years) </name><val> 37.0 </val></parm>
<var><name> Age(Years) </name></var>
<var><name> Index </name><val> 2 </val></var>

<var><name> IsYoung </name><val> FALSE </val></var>
<var><name> IsYoungish </name><val> FALSE </val></var>
<var><name> IsMiddleaged </name><val> TRUE </val></var>
<var><name> IsOld </name><val> FALSE </val></var>
<var><name> IsVeryOld </name><val> FALSE </val></var>

</variables>

<definitions>

<block><name> Parms </name>

<onjustchanged><name> InitialAge(Years) </name>
<call> Update </call>
<call> IndexEverybody </call>
<call> Context.Size </call>
</onjustchanged>

</block>

<block><name> Update </name>
</block>

<block><name> Normalize </name>

<call> NormalizeEverybody </call>
<call> Update </call>

</block>

<block><name> Calc </name>

<def>
<name> Age(Years) </name>
<val> InitialAge(Years) + ( ( System.X - System.InitialX ) / 525600.0 ) </val>
</def>

</block>

<block><name> IndexEverybody </name>

<testcase>

<case><test> InitialAge(Years) LT ( ( YOUNG + YOUNGISH ) / 2 ) </test>
<def><name> Index </name><val> 0 </val></def>
</case>

<case><test> InitialAge(Years) LT ( ( YOUNGISH + MIDDLEAGED ) / 2 ) </test>
<def><name> Index </name><val> 1 </val></def>
</case>

<case><test> InitialAge(Years) LT ( ( MIDDLEAGED + OLD ) / 2 ) </test>
<def><name> Index </name><val> 2 </val></def>
</case>

<case><test> InitialAge(Years) LT ( ( OLD + VERYOLD ) / 2 ) </test>
<def><name> Index </name><val> 3 </val></def>
</case>

<case><test> TRUE </test>
<def><name> Index </name><val> 4 </val></def>
</case>

</testcase>

</block>

<block><name> NormalizeEverybody </name>

<testcase>

<case><test> Index EQ 0 </test>
<def><name> InitialAge(Years) </name><val> YOUNG </val></def>
</case>

<case><test> Index EQ 1 </test>
<def><name> InitialAge(Years) </name><val> YOUNGISH </val></def>
</case>

<case><test> Index EQ 2 </test>
<def><name> InitialAge(Years) </name><val> MIDDLEAGED </val></def>
</case>

<case><test> Index EQ 3 </test>
<def><name> InitialAge(Years) </name><val> OLD </val></def>
</case>

<case><test> Index EQ 4 </test>
<def><name> InitialAge(Years) </name><val> VERYOLD </val></def>
</case>

</testcase>

</block>

</definitions>
</structure>
