<structure><name> RightHeartPumping-Diastole </name>

<variables>

<var><name> EDV </name></var>
<var><name> EDP </name></var>
<var><name> TMP </name></var>

<parm><name> A-Basic </name><val> 0.00026 </val></parm>
<var><name> A </name></var>
<parm><name> n </name><val> 2.0 </val></parm>

<parm><name> Stiffness </name><val> 1.0 </val></parm>

<var><name> TricuspidStenosis </name></var>

<var><name> TMP_Power </name></var>

</variables>

<definitions>

<block><name> Calc </name>

<def>
<name> TricuspidStenosis </name>
<val> TricuspidValve-Stenosis.Effect </val>
</def>

<def>
<name> EDP </name>
<val>
( TricuspidStenosis * RightAtrium.Pressure )
+ ( ( 1 - TricuspidStenosis ) * Pericardium-Cavity.Pressure )
</val>
</def>

<def><name> EDP </name><val> RightAtrium.Pressure </val></def>
<def><name> TMP </name><val> EDP - Pericardium-Cavity.Pressure </val></def>
<def><name> A </name><val> Stiffness * A-Basic </val></def>

<call> V=f(P) </call>

</block>

<block><name> P=f(V) </name>
<def><name> TMP_Power </name><val> ( EDV ) ^ ( n ) </val></def>
<def><name> TMP </name><val> A * TMP_Power </val></def>
</block>

<block><name> V=f(P) </name>
<def><name> EDV </name><val> ( TMP / A ) ^ ( 1.0 / n ) </val></def>
</block>

</definitions>
</structure>

End
