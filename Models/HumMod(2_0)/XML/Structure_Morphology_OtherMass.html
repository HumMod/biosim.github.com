<structure><name> OtherMass </name>

<variables>

<constant>
<name> MALE_SMALL </name>
<val> 19.0 </val>
</constant>

<constant>
<name> MALE_NORMAL </name>
<val> 25.0 </val>
</constant>

<constant>
<name> MALE_LARGE </name>
<val> 32.0 </val>
</constant>

<constant>
<name> MALE_VERYLARGE </name>
<val> 46.0 </val>
</constant>

<constant>
<name> FEMALE_SMALL </name>
<val> 15.0 </val>
</constant>

<constant>
<name> FEMALE_NORMAL </name>
<val> 20.0 </val>
</constant>

<constant>
<name> FEMALE_LARGE </name>
<val> 28.0 </val>
</constant>

<constant>
<name> FEMALE_VERYLARGE </name>
<val> 36.0 </val>
</constant>

<parm><name> Mass(kG) </name><val> 25.0 </val></parm>

<var><name> Mass </name></var>
<var><name> Index </name><val> 1 </val></var>

<var><name> IsSmall </name><val> FALSE </val></var>
<var><name> IsNormal </name><val> TRUE </val></var>
<var><name> IsLarge </name><val> FALSE </val></var>
<var><name> IsVeryLarge </name><val> FALSE </val></var>

</variables>

<definitions>

<block><name> Parms </name>

<onjustchanged><name> Mass(kG) </name>
<call> Update </call>

<if><test> Gender.IsFemale </test>
<true><call> IndexFemale </call></true>
<false><call> IndexMale </call></false>
</if>

<call> Context.Size </call>
</onjustchanged>

</block>

<block><name> Update </name>

<def>
<name> Mass </name>
<val> 1000 * Mass(kG) </val>
</def>

</block>

<block><name> Normalize </name>

<if><test> Gender.IsFemale </test>
<true><call> NormalizeFemale </call></true>
<false><call> NormalizeMale </call></false>
</if>

<call> Update </call>

</block>

<block><name> IndexMale </name>

<testcase>

<case><test> Mass(kG) LT ( ( MALE_SMALL + MALE_NORMAL ) / 2 ) </test>
<def><name> Index </name><val> 0 </val></def>
</case>

<case><test> Mass(kG) LT ( ( MALE_NORMAL + MALE_LARGE ) / 2 ) </test>
<def><name> Index </name><val> 1 </val></def>
</case>

<case><test> Mass(kG) LT ( ( MALE_LARGE + MALE_VERYLARGE ) / 2 ) </test>
<def><name> Index </name><val> 2 </val></def>
</case>

<case><test> TRUE </test>
<def><name> Index </name><val> 3 </val></def>
</case>

</testcase>

</block>

<block><name> IndexFemale </name>

<testcase>

<case><test> Mass(kG) LT ( ( FEMALE_SMALL + FEMALE_NORMAL ) / 2 ) </test>
<def><name> Index </name><val> 0 </val></def>
</case>

<case><test> Mass(kG) LT ( ( FEMALE_NORMAL + FEMALE_LARGE ) / 2 ) </test>
<def><name> Index </name><val> 1 </val></def>
</case>

<case><test> Mass(kG) LT ( ( FEMALE_LARGE + FEMALE_VERYLARGE ) / 2 ) </test>
<def><name> Index </name><val> 2 </val></def>
</case>

<case><test> TRUE </test>
<def><name> Index </name><val> 3 </val></def>
</case>

</testcase>

</block>

<block><name> NormalizeMale </name>

<testcase>

<case><test> Index EQ 0 </test>
<def><name> Mass(kG) </name><val> MALE_SMALL </val></def>
</case>

<case><test> Index EQ 1 </test>
<def><name> Mass(kG) </name><val> MALE_NORMAL </val></def>
</case>

<case><test> Index EQ 2 </test>
<def><name> Mass(kG) </name><val> MALE_LARGE </val></def>
</case>

<case><test> Index EQ 3 </test>
<def><name> Mass(kG) </name><val> MALE_VERYLARGE </val></def>
</case>

</testcase>

</block>

<block><name> NormalizeFemale </name>

<testcase>

<case><test> Index EQ 0 </test>
<def><name> Mass(kG) </name><val> FEMALE_SMALL </val></def>
</case>

<case><test> Index EQ 1 </test>
<def><name> Mass(kG) </name><val> FEMALE_NORMAL </val></def>
</case>

<case><test> Index EQ 2 </test>
<def><name> Mass(kG) </name><val> FEMALE_LARGE </val></def>
</case>

<case><test> Index EQ 3 </test>
<def><name> Mass(kG) </name><val> FEMALE_VERYLARGE </val></def>
</case>

</testcase>

</block>

</definitions>
</structure>

End
